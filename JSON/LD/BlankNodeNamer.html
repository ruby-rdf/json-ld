def get_sym(old = "")
  old = old.to_s.sub(/_:/, '')
  if !old.empty? && key?(old)
    self[old]
  elsif !old.empty?
    @num += 1
    # puts "allocate #{@prefix + (@num - 1).to_s} to #{old.inspect}"
    self[old] = @prefix + (@num - 1).to_s
  else
    # Not referenced, just return a new unique value
    @num += 1
    # puts "allocate #{@prefix + (@num - 1).to_s} to #{old.inspect}"
    @prefix + (@num - 1).to_s
  end
end
